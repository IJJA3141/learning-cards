string getFile(string fileName, string spliter) {
    fstream file;
    string localeVocList;
    file.open(fileName, ios::in);
    if (file.is_open())
    {
        string line;
        while (getline(file, line))
        {
            localeVocList = line;
        }
        file.close();
    }

    return localeVocList;
}

void store(string fileName, string text)
{
    fstream file;
    file.open(fileName, ios::out);
    if (file.is_open())
    {
        file << text;
        file.close();
    }
}
